# 上位机 ASR + NLU 推理服务 (Python 3.8 兼容)
# 适用于部署环境只有 Python 3.8 的场景
#
# 安装方法:
#   pip install -r requirements-server-py38.txt
#
# 启动方法:
#   python -m asr.server --serve --host 0.0.0.0 --port 8000
#   python -m nlu.server --serve --host 0.0.0.0 --port 8001
#
# 或通过 run.py 统一启动 (需配合 --server-only):
#   python run.py --server-only

# ─── 推理引擎 ───
onnxruntime>=1.19.0,<=1.19.2    # 1.19.2 = 最后支持 3.8 的版本, 支持 MatMulNBits/INT4
numpy>=1.24.0,<2.0              # 2.0+ 需要 Python 3.9+

# ─── ASR 分词器 ───
tokenizers>=0.19.0,<=0.20.3     # Tokenizer.from_file() 加载 Qwen3-ASR BPE

# ─── NLU 分词器 ───
transformers>=4.36.0,<=4.38.2   # AutoTokenizer + T5Tokenizer
sentencepiece>=0.1.97,<0.3      # T5Tokenizer spiece.model 后备

# ─── 音频处理 ───
soundfile>=0.12.1,<0.13         # WAV 音频读取
librosa>=0.10.0,<0.11           # mel 滤波器组 + 重采样 (0.11+ 需要 3.9)

# ─── HTTP 服务 ───
fastapi>=0.100.0,<=0.104.1      # 0.104.x = 最后支持 3.8 的版本
uvicorn>=0.20.0,<0.28           # ASGI 服务器
python-multipart>=0.0.6,<0.1   # FastAPI File/Form 上传依赖
pydantic>=1.10.0,<3.0           # FastAPI 数据模型

# ─── 音频处理子依赖 (锁定 3.8 兼容范围) ───
scipy>=1.10.0,<1.11             # librosa 依赖, 1.11+ 需要 3.9
